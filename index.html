            // Typescript program
            
            class ControleDeAudio {
              volume: number = 2;
          
              aumentarVolume(): void {
                  if (this.volume <= 10){
                      this.volume += 1;
          
                  }
              }
          
              diminuirVolume(): void {
                  if (this.volume > 0){
                      this.volume -= 1;
                  }
              }
          
              getVolume(): number {
                  return this.volume;
              }
            }
          
            let audio: ControleDeAudio = new ControleDeAudio();
            
            console.log(audio.getVolume());
            
            while(audio.volume < 10){
                audio.aumentarVolume();
            }
            
            console.log(audio.getVolume());
            
            
            while(audio.volume > 0){
                audio.diminuirVolume();
            }
            
            console.log(audio.getVolume());
